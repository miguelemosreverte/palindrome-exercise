# Definimos la función para calcular los números de Fibonacci
def fibonacci(n):
    secuencia = []
    a, b = 0, 1
    for _ in range(n):
        secuencia.append(a)
        a, b = b, a + b
    return secuencia

# Calculamos los primeros 10 números de Fibonacci
numeros_fibonacci = fibonacci(10)

# Mostramos los resultados
print(numeros_fibonacci)
